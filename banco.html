🗄️ Banco de Dados

Schema e modelo de dados para processar pagamentos multi-gateway

Visão Geral

O schema de banco de dados é agnóstico ao integrador — não importa se a transação foi processada via Getnet, Mercado Pago ou Pagar.me, a estrutura é a mesma. Campos específicos do integrador ficam em colunas JSON/JSONB ou tabelas auxiliares.

📋 Stack de DB PostgreSQL (Supabase ou RDS) · Drizzle ORM para migrations e type-safety · Tipos nativos UUID, JSONB, TIMESTAMP WITH TIME ZONE

Tabelas Principais — Schema SQL

1. customers — Clientes/Compradores

CREATE TABLE customers (
    id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    email         TEXT NOT NULL,
    first_name    VARCHAR(100),
    last_name     VARCHAR(100),
    document_type VARCHAR(10),        -- CPF, CNPJ
    document_number VARCHAR(20),      -- sem pontuação
    phone_number  VARCHAR(20),
    created_at    TIMESTAMPTZ DEFAULT now(),
    updated_at    TIMESTAMPTZ DEFAULT now(),

    UNIQUE(email)
);

CREATE INDEX idx_customers_doc ON customers(document_type, document_number);

2. transactions — Transações (Pagamentos)

CREATE TABLE transactions (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    customer_id     UUID REFERENCES customers(id),

    -- Identificação do pedido
    order_id        VARCHAR(100) NOT NULL,     -- ID do pedido no nosso sistema
    idempotency_key VARCHAR(64) NOT NULL UNIQUE, -- previne dupla cobrança

    -- Gateway/Integrador
    provider        VARCHAR(20) NOT NULL,     -- 'getnet', 'mercado-pago', 'pagarme'
    provider_payment_id  VARCHAR(255),        -- ID retornado pelo gateway
    provider_transaction_id VARCHAR(255),    -- transaction_id do gateway

    -- Dados do pagamento
    amount          INTEGER NOT NULL,          -- valor em CENTAVOS (R$ 100,00 = 10000)
    currency        VARCHAR(3) DEFAULT 'BRL',
    payment_method  VARCHAR(20) NOT NULL,     -- 'credit_card', 'debit_card', 'pix', 'boleto'
    installments    INTEGER DEFAULT 1,
    transaction_type VARCHAR(20) DEFAULT 'FULL', -- FULL, INSTALL_NO_INTEREST, etc.

    -- Status
    status          VARCHAR(20) NOT NULL DEFAULT 'PENDING',
    -- PENDING, AUTHORIZED, APPROVED, DENIED, CANCELED, ERROR, WAITING, REFUNDED
    status_detail   TEXT,                     -- mensagem de erro/reason_message
    reason_code     VARCHAR(10),              -- código de reason (ex: '00')

    -- Cartão (quando aplicável)
    card_brand      VARCHAR(20),             -- VISA, MASTERCARD, ELO, AMEX, HIPER
    card_last4      CHAR(4),                  -- últimos 4 dígitos
    card_token_id   UUID,                     -- FK para card_tokens (se tokenizado)
    authorization_code VARCHAR(20),

    -- Pix (quando aplicável)
    pix_qr_code     TEXT,                     -- string "copia e cola"
    pix_expires_at  TIMESTAMPTZ,
    pix_paid_at     TIMESTAMPTZ,

    -- Metadata
    soft_descriptor VARCHAR(50),
    provider_response JSONB,                  -- resposta completa do gateway
    provider_request  JSONB,                  -- request enviado (sem PAN/CVV)
    metadata       JSONB DEFAULT '{}',       -- dados adicionais custom

    -- Antifraude
    device_ip       INET,
    device_id       UUID,
    finger_print    VARCHAR(255),

    -- Timestamps
    created_at      TIMESTAMPTZ DEFAULT now(),
    updated_at      TIMESTAMPTZ DEFAULT now(),
    authorized_at   TIMESTAMPTZ,
    received_at     TIMESTAMPTZ,

    UNIQUE(order_id, provider)
);

CREATE INDEX idx_txn_status ON transactions(status);
CREATE INDEX idx_txn_provider ON transactions(provider);
CREATE INDEX idx_txn_order ON transactions(order_id);
CREATE INDEX idx_txn_customer ON transactions(customer_id);
CREATE INDEX idx_txn_created ON transactions(created_at DESC);
CREATE INDEX idx_txn_provider_pid ON transactions(provider_payment_id);

3. card_tokens — Cartões Tokenizados (Cofre)

CREATE TABLE card_tokens (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    customer_id     UUID REFERENCES customers(id),
    provider        VARCHAR(20) NOT NULL,    -- em qual gateway foi tokenizado
    provider_token  VARCHAR(255) NOT NULL,   -- token retornado pelo gateway

    card_brand      VARCHAR(20) NOT NULL,
    card_last4      CHAR(4) NOT NULL,
    expiration_month VARCHAR(2) NOT NULL,
    expiration_year VARCHAR(4) NOT NULL,
    cardholder_name VARCHAR(100),

    is_active       BOOLEAN DEFAULT true,
    created_at      TIMESTAMPTZ DEFAULT now(),
    updated_at      TIMESTAMPTZ DEFAULT now(),

    UNIQUE(provider, provider_token)
);

CREATE INDEX idx_card_tokens_customer ON card_tokens(customer_id);

4. webhook_events — Log de Webhooks Recebidos

CREATE TABLE webhook_events (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    transaction_id  UUID REFERENCES transactions(id),

    provider        VARCHAR(20) NOT NULL,    -- 'getnet', 'mercado-pago', 'pagarme'
    event_type      VARCHAR(50),            -- 'payment.approved', 'payment.denied', etc.
    provider_event_id VARCHAR(255),         -- ID do evento no gateway

    -- Dados do webhook
    raw_body        JSONB NOT NULL,         -- payload completo recebido
    raw_headers     JSONB,                  -- headers (para validação de assinatura)
    query_params    JSONB,                  -- para Getnet que usa HTTP GET

    -- Status do processamento
    processed       BOOLEAN DEFAULT false,
    processed_at    TIMESTAMPTZ,
    processing_error TEXT,
    retry_count     INTEGER DEFAULT 0,

    received_at     TIMESTAMPTZ DEFAULT now(),

    UNIQUE(provider, provider_event_id)
);

CREATE INDEX idx_webhook_txn ON webhook_events(transaction_id);
CREATE INDEX idx_webhook_unprocessed ON webhook_events(processed) WHERE processed = false;

5. refunds — Estornos/Cancelamentos

CREATE TABLE refunds (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    transaction_id  UUID REFERENCES transactions(id),

    provider        VARCHAR(20) NOT NULL,
    provider_refund_id VARCHAR(255),

    amount          INTEGER NOT NULL,       -- valor do estorno em centavos
    status          VARCHAR(20) DEFAULT 'PENDING', -- PENDING, APPROVED, DENIED, ERROR
    reason          TEXT,

    idempotency_key VARCHAR(64) NOT NULL UNIQUE,

    provider_response JSONB,
    created_at      TIMESTAMPTZ DEFAULT now(),
    processed_at    TIMESTAMPTZ
);

CREATE INDEX idx_refund_txn ON refunds(transaction_id);

6. split_payments — Divisão de Valores (Split)

CREATE TABLE split_payments (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    transaction_id  UUID REFERENCES transactions(id),

    receiver_id     VARCHAR(255) NOT NULL,  -- ID do sub-vendedor no gateway
    receiver_name   VARCHAR(200),

    amount          INTEGER NOT NULL,        -- valor em centavos
    percentage      DECIMAL(5,2),           -- ou percentual
    fee_type        VARCHAR(20),            -- 'PERCENTAGE' ou 'FIXED'

    status          VARCHAR(20) DEFAULT 'PENDING',

    created_at      TIMESTAMPTZ DEFAULT now()
);

CREATE INDEX idx_split_txn ON split_payments(transaction_id);

7. idempotency_keys — Controle de Idempotência

CREATE TABLE idempotency_keys (
    key             VARCHAR(64) PRIMARY KEY,
    endpoint        VARCHAR(100) NOT NULL,   -- qual endpoint usou esta chave
    request_body    JSONB,                   -- body enviado (hash para comparação)
    response_body   JSONB,                   -- resposta do gateway
    status_code     INTEGER,
    created_at      TIMESTAMPTZ DEFAULT now(),
    expires_at      TIMESTAMPTZ DEFAULT now() + INTERVAL '24 hours'
);

CREATE INDEX idx_idempotency_expires ON idempotency_keys(expires_at);

Fluxo de Dados — Cartão de Crédito

1. Frontend → Tokeniza cartão via SDK do gateway → recebe card_token
2. Frontend → POST /api/charge { provider, card_token, amount, customer }
3. Backend  → INSERT INTO transactions (status='PENDING', idempotency_key)
4. Backend  → Chama API do gateway → recebe payment_id + status
5. Backend  → UPDATE transactions SET status='APPROVED', provider_payment_id
6. Gateway  → Webhook → POST /api/webhook/{provider}
7. Backend  → INSERT INTO webhook_events → UPDATE transactions (status final)

Fluxo de Dados — Pix

1. Frontend → POST /api/pix { provider, amount, customer }
2. Backend  → INSERT INTO transactions (status='WAITING', idempotency_key)
3. Backend  → Chama API do gateway → recebe qr_code + payment_id
4. Backend  → UPDATE transactions SET pix_qr_code, pix_expires_at
5. Frontend → Exibe QR Code para o cliente
6. Cliente  → Paga via app do banco
7. Gateway  → Webhook → POST /api/webhook/{provider}
8. Backend  → INSERT INTO webhook_events
9. Backend  → UPDATE transactions SET status='APPROVED', pix_paid_at

Triggers Úteis

-- Auto-update updated_at em todas as tabelas
CREATE OR REPLACE FUNCTION update_updated_at()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = now();
    RETURN NEW;
END;
$$ language 'plpgsql';

CREATE TRIGGER txn_updated_at BEFORE UPDATE ON transactions
    FOR EACH ROW EXECUTE FUNCTION update_updated_at();

CREATE TRIGGER customer_updated_at BEFORE UPDATE ON customers
    FOR EACH ROW EXECUTE FUNCTION update_updated_at();

-- Log de mudança de status
CREATE TABLE transaction_status_log (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    transaction_id  UUID REFERENCES transactions(id),
    old_status      VARCHAR(20),
    new_status      VARCHAR(20),
    changed_at      TIMESTAMPTZ DEFAULT now()
);

CREATE OR REPLACE FUNCTION log_status_change()
RETURNS TRIGGER AS $$
BEGIN
    IF NEW.status IS DISTINCT FROM OLD.status THEN
        INSERT INTO transaction_status_log (transaction_id, old_status, new_status)
        VALUES (NEW.id, OLD.status, NEW.status);
    END IF;
    RETURN NEW;
END;
$$ language 'plpgsql';

CREATE TRIGGER txn_status_log BEFORE UPDATE ON transactions
    FOR EACH ROW EXECUTE FUNCTION log_status_change();

Retenção e Particionamento

⚠️ Estratégia de Retenção
  • transactions: Retenção indefinida (registro fiscal/contábil)
  • webhook_events: Retenção 90 dias (para troubleshooting). Após isso, arquivar
  • idempotency_keys: Expiram em 24h (cleanup automático via job)
  • transaction_status_log: Retenção 12 meses
  • Para volumes altos, particionar transactions por mês (PARTITION BY RANGE (created_at))

Drizzle ORM — Schema (TypeScript)

// drizzle/schema.ts
import { pgTable, uuid, varchar, integer, boolean, jsonb, timestamp, text, char, inet, decimal } from 'drizzle-orm/pg-core';

export const customers = pgTable('customers', {
  id: uuid('id').defaultRandom().primaryKey(),
  email: varchar('email', { length: 255 }).notNull(),
  first_name: varchar('first_name', { length: 100 }),
  last_name: varchar('last_name', { length: 100 }),
  document_type: varchar('document_type', { length: 10 }),
  document_number: varchar('document_number', { length: 20 }),
  phone_number: varchar('phone_number', { length: 20 }),
  created_at: timestamp('created_at', { withTimezone: true }).defaultNow(),
  updated_at: timestamp('updated_at', { withTimezone: true }).defaultNow(),
});

export const transactions = pgTable('transactions', {
  id: uuid('id').defaultRandom().primaryKey(),
  customer_id: uuid('customer_id').references(() => customers.id),
  order_id: varchar('order_id', { length: 100 }).notNull(),
  idempotency_key: varchar('idempotency_key', { length: 64 }).notNull().unique(),
  provider: varchar('provider', { length: 20 }).notNull(),
  provider_payment_id: varchar('provider_payment_id', { length: 255 }),
  amount: integer('amount').notNull(),
  currency: varchar('currency', { length: 3 }).default('BRL'),
  payment_method: varchar('payment_method', { length: 20 }).notNull(),
  installments: integer('installments').default(1),
  status: varchar('status', { length: 20 }).notNull().default('PENDING'),
  provider_response: jsonb('provider_response'),
  created_at: timestamp('created_at', { withTimezone: true }).defaultNow(),
  updated_at: timestamp('updated_at', { withTimezone: true }).defaultNow(),
  // ... campos omitidos para brevidade
});
